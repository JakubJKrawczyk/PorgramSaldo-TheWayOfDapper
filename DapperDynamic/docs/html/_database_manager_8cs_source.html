40 {
41 if(reopenConnection)
42 try

44 _connection.Open();

46 catch (InvalidOperationException){}
47
48 using var transaction = _connection.BeginTransaction();

50 {
51 bool result = statement.Invoke();

185 public bool Insert(InsertQuery insertQuery)
186 {
187 if(!_isTableExists(insertQuery.TableDisplayName)) throw new ArgumentException("Table does not exist");
188 var tablerealname = _connection.QueryFirstOrDefault(
189 "SELECT `tablerealname` FROM `usertables` WHERE `tabledisplayname` = @tablename",

191 if(tablerealname is null) throw new ArgumentException("Table does not exist");
192 var columns = _connection.Query(
193 "SELECT `realname`, `displayname`, `csharptype` FROM `usertablescolumns` WHERE `tablerealname` = @tablename",
194 new { tablename = tablerealname.tablerealname });
195 if(columns is null) throw new ArgumentException("Table does not exist");
196 var values = new Dictionary<string, object>();
197 foreach(var column in columns)
198 {
199 if(!insertQuery.Parameters.ContainsKey(column.displayname))
200 throw new ArgumentException("Missing value for column");
201 values.Add(column.realname, insertQuery.Parameters[column.displayname]);
202 }
203 string sql = $"INSERT INTO `{tablerealname.tablerealname}` (" +
204 string.Join(", ", values.Keys) +
205 ") VALUES (" +
206 string.Join(", ", values.Keys.Select(k => "@" + k)) +
207 ")";
208 return HandleInTransaction(() =>
209 {
210 int result = _connection.Execute(sql, values);
211 return result == 1;

285 public IEnumerable<dynamic> Select(SelectQuery query)
286 {
287 var toTranslate = ((IStatement)query).GetNamesToTranslate();
288 var dictionary = new Dictionary<string, string>();
289 foreach(var (name, type) in toTranslate)
290 {
291 if(type == NameType.Table)
292 {
293 dictionary.Add(name, _connection.QueryFirst(
294 "SELECT `tablerealname` AS trn FROM `usertables` WHERE `tabledisplayname` = @name",
295 new { name })
296 .trn);
297 continue;
298 }
299
300 if (name.Contains("."))
301 {
302 var split = name.Split('.');
303 dictionary.Add(split[1], _connection.QueryFirst(
304 "SELECT `realname` AS crn" +
305 " FROM `usertablescolumns` INNER JOIN usertables u on usertablescolumns.tablerealname = u.tablerealname" +
306 " WHERE `tabledisplayname` = @table AND `displayname` = @name",
307 new { table = dictionary[split[0]], name = split[1] })
308 .crn);
309 }
310 else
311 {
312 var tables = query.GetInvolvedTables();
313 foreach(var table in tables)
314 {
315 var column = _connection.QueryFirstOrDefault(
316 "SELECT `realname` AS crn" +
317 " FROM `usertablescolumns` INNER JOIN usertables u on usertablescolumns.tablerealname = u.tablerealname" +
318 " WHERE `tabledisplayname` = @table AND `displayname` = @name",
319 new { table = table, name = name });
320 if(column is null) continue;
321 dictionary.Add(name, column.crn);
322 break;
323 }
324 }
325 }
326 var statement = ((IStatement)query).GetStatement(dictionary);
327 return _connection.Query(statement.Item1, statement.Item2);
